240      public void setupTimeline(double[] times) {
241          if (this.modelStartTimes == null) {
242              this.modelStartTimes = new double[numIntervals];
243          } else {
244              Arrays.fill(this.modelStartTimes, 0.0);
245          }
246          if (times != null) {
247              if (times.length != this.numIntervals) {
248                  throw new IllegalArgumentException("grids has the wrong dimension " + times.length + ", not matching number of intervals " + this.numIntervals + "!");
249              }
250              this.modelStartTimes = times;
251          } else {
252              for (int idx = 1; idx <= numIntervals - 1 ; idx++) {
253                  modelStartTimes[idx] = idx * (gridEnd / numIntervals);
254              }
255          }
256      }

263      final double p(int model, double t) {
264          double eAt = Math.exp(A * (t - modelStartTimes[model]));
265          return p(eAt);
266      }
267  
268      private double p(double eAt) {
269          double eAt1B = eAt * (1.0 + B);
270          return (lambda + mu + psi - A * ((eAt1B - (1.0 - B)) / (eAt1B + (1.0 - B)))) / (2.0 * lambda);
271      }
272  
273      double logQ(int model, double time) {
274          double At = A * (time - modelStartTimes[model]);
275          double eAt = Math.exp(At);
276          double sqrtDenominator = g1(eAt);
277          return At + log4 - 2 * Math.log(sqrtDenominator); // TODO log4 (additive constant) is not needed since we always see logQ(a) - logQ(b)
278      }

280      private static final double log4 = Math.log(4.0);
281      
282      // Named as per Gavryushkina et al 2014
283      private static double computeA(double lambda, double mu, double psi) {
284          return Math.abs(Math.sqrt(Math.pow(lambda - mu - psi, 2.0) + 4.0 * lambda * psi));  // TODO Why do we have Math.abs(always positive #)?
285      }
286  
287      // Named as per Gavryushkina et al 2014
288      private static double computeB(double lambda, double mu, double psi, double rho, double A, double pPrevious) {
289          return ((1.0 - 2.0 * (1 - rho) * pPrevious) * lambda + mu + psi)/A;
290      }

292      @Override
293      public double logConditioningProbability(int model) {
294          double logP = 0.0;
295          if ( conditionOnSurvival ) {
296              double origin = originTime.getParameterValue(0);
297              double[] modelBreakPoints = getBreakPoints();
298  
299              double segmentIntervalEnd = modelBreakPoints[model];
300  
301              while (origin >= segmentIntervalEnd) { // TODO Maybe it's >= ?
302                  ++model;
303                  updateLikelihoodModelValues(model);
304                  segmentIntervalEnd = modelBreakPoints[model];
305              }
306  
307              logP -= Math.log(1.0 - p(model, origin));
308          }
309          return logP;
310      }

328      @Override
329      public double processModelSegmentBreakPoint(int model, double intervalStart, double intervalEnd, int nLineages) {
330          double lnL = nLineages * (logQ(model, intervalEnd) - logQ(model, intervalStart));
331          if ( samplingProbability.getValue(model + 1) > 0.0 && samplingProbability.getValue(model + 1) < 1.0) {
332              // Add in probability of un-sampled lineages
333              // We don't need this at t=0 because all lineages in the tree are sampled
334              // TODO: check if we're right about how many lineages are actually alive at this time. Are we inadvertently over-counting or under-counting due to samples added at this _exact_ time?
335              lnL += nLineages * Math.log(1.0 - samplingProbability.getValue(model + 1));
336          }
337          this.savedLogQ = Double.NaN;
338          return lnL;
339      }

342          lambda = birthRate.getParameterValue(model);
343          mu = deathRate.getParameterValue(model);
344          psi = serialSamplingRate.getParameterValue(model);
345          r = treatmentProbability.getParameterValue(model);
346          rho = samplingProbability.getParameterValue(model);

348          this.savedLogQ = Double.NaN;

351      @Override
352      public void updateLikelihoodModelValues(int model) {
353          
354          modelStartTime = modelStartTimes[model];
355  
356          if (model == 0) {
357              previousP = 1.0;
358          } else{
359              previousP = p(model - 1, modelStartTime);
360          }
361  
362          updateParameterValues(model);
363  
364  
365          A = computeA(lambda, mu, psi);
366          B = computeB(lambda, mu, psi, rho, A, previousP);
367      }

369      @Override
370      public void updateGradientModelValues(int model) {
371  
372          modelStartTime = modelStartTimes[model];
373  
374          if (model == 0) {
375              previousP = 1.0;
376          } else{
377              previousP = p(eAt_End);
378          }
379  
380          updateParameterValues(model);
381  
382          A = computeA(lambda, mu, psi);
383          B = computeB(lambda, mu, psi, rho, A, previousP);
384          
385          this.savedQ = Double.NaN;
386          this.partialQKnown = false;
387  
388          dACompute(dA);
389          
390          dBCompute(model, dB);
391  
392          System.arraycopy(dPModelEnd, 0, dPModelEnd_prev, 0, numIntervals * 4);
393  
394          if (numIntervals > 1 && model < numIntervals - 1) {
395  
396              double end = modelStartTimes[model + 1];
397              double start = modelStartTimes[model];
398              eAt_End = Math.exp(A * (end - start));
399              dPCompute(model, end, start, eAt_End, dPModelEnd, dG2);
400          }
401          
402          computedBCurrent = true;
403      }

405      @Override
406      public double processInterval(int model, double tYoung, double tOld, int nLineages) {
407          double logQ_young;
408          double logQ_old = logQ(model, tOld);
409          if (!Double.isNaN(this.savedLogQ)) {
410              logQ_young = this.savedLogQ;
411          } else {
412              logQ_young = logQ(model, tYoung);
413          }
414          this.savedLogQ = logQ_old;
415          return nLineages * (logQ_old - logQ_young);
416      }

423      @Override
424      public double processSampling(int model, double tOld) {
425  
426          double logSampProb;
427  
428          boolean sampleIsAtEventTime = tOld == modelStartTimes[model];
429          boolean samplesTakenAtEventTime = rho > 0;
430  
431          if (sampleIsAtEventTime && samplesTakenAtEventTime) {
432              logSampProb = Math.log(rho);
433              if (model > 0) {
434                  logSampProb = Math.log(rho) + Math.log(r + ((1.0 - r) * previousP));
435              }
436          } else {
437              double logPsi = Math.log(psi);
438              logSampProb = logPsi + Math.log(r + (1.0 - r) * p(model,tOld));
439          }
440  
441          return logSampProb;
442      }

444      @Override
445      public double processOrigin(int model, double rootAge) {
446          if (originTime.getValue(0) < rootAge) {
447              return Double.NaN;
448          } else {
449              double[] modelBreakPoints = getBreakPoints();
450              double logL = 0.0;
451  
452              double origin = originTime.getValue(0);
453              double intervalStart = rootAge;
454              double segmentIntervalEnd = modelBreakPoints[model];
455  
456              while (origin >= segmentIntervalEnd) { // TODO Maybe it's >= ?
457                  logL += processModelSegmentBreakPoint(model, intervalStart, segmentIntervalEnd, 1);
458                  intervalStart = segmentIntervalEnd;
459                  ++model;
460                  updateLikelihoodModelValues(model);
461                  segmentIntervalEnd = modelBreakPoints[model];
462              }
463              if (intervalStart < origin) {
464                  logL += logQ(model, origin) - logQ(model, intervalStart);
465              }
466  
467              return logL;
468          }
469      }

501      final double g1(double eAt) {
502          return  (1 + B) * eAt + (1 - B);
503      }
504  
505      final double g2(double G1) {
506          return A * (1 - 2 * (1 - B) / G1);
507      }

615      void dPCompute(int model, double t, double intervalStart, double eAt, double[] dP, double[] dG2) {
616  
617          double G1 = g1(eAt);
618  
619          double term1 = -A / lambda * ((1 - B) * (eAt - 1) + G1) / (G1 * G1);
620  
621          for (int k = 0; k  < model; k ++) {
622              for (int p = 0; p < 4; p++) {
623                  dP[k * 4 + p] = term1 * dB[k * 4 + p];
624              }
625          }
626  
627          for (int p = 0; p < 3; ++p) { // TODO Only dG1[1] and dG2[2] are used
628              double term2 = eAt * (1 + B) * dA[p] * (t - intervalStart) + (eAt - 1) * dB[model * 4 + p];
629              dG2[p] = dA[p] - 2 * (G1 * (dA[p] * (1 - B) - dB[model * 4 + p] * A) - (1 - B) * term2 * A) / (G1 * G1);
630          }
631  
632          double G2 = g2(G1);
633  
634          dP[model * 4 + 0] = (-mu - psi - lambda * dG2[0] + G2) / (2 * lambda * lambda);
635          dP[model * 4 + 1] = (1 - dG2[1]) / (2 * lambda);
636          dP[model * 4 + 2] = (1 - dG2[2]) / (2 * lambda);
